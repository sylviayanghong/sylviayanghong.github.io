实现addClass

        .abc{
            color: #f00;
        }
    </style>
</head>
<body>
    <h3 class="test">好好学习</h3>
    <button class="btn">添加</button>

    <script>

        var h = document.querySelector('h3');

        function addClass(dom,str){
            var newclassName = '';
            if(dom.className){
                var className = dom.className;
                newclassName = className + ' ' + str;
            }else{
                newclassName = str;
            }
            dom.className = newclassName;
        }
        document.querySelector('.btn').onclick = function(){
            addClass(h,'abc');
        }


    </script>
</body>

    <script>

        function siblings(dom) {
            var newArr = [];
            var par = dom.parentNode;  //得到s的父节点
            var child= par.children;  //得到s的全部子节点（元素节点）
            for(var i = 0;i<child.length;i++){
                if(child[i] !== dom){
                    newArr.push(child[i]);
                }
            }
            return newArr;
        }

        var oLi = document.querySelectorAll(".test li");
        for(var i = 0;i<oLi.length;i++){
            oLi[i].onclick = function(){
                console.log(siblings(this));
            }
        }
