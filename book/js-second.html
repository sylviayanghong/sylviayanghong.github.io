倒计时

<h3 class="time"></h3>

    <script>

        function fun(){
            var target = new Date('2017-3-31 17:00');
            var targetTime = target.getTime();

            var dateNow = new Date();
            var nowTime = dateNow.getTime();

            var sub = targetTime - nowTime;

            var day = Math.floor(sub / (1000*60*60*24));
            var subDay = Math.floor(sub%(1000*60*60*24));

            var hours = Math.floor(subDay/(1000*60*60));
            var subTime = Math.floor(subDay%(1000*60*60));

            var minutes = Math.floor(subTime/(1000*60));
            var subMinutes = Math.floor(subTime%(1000*60));

            var seconds = Math.floor(subMinutes/(1000));
            var hTime = '距休息日还有：' + day + '天' + hours + '小时' + minutes + '分' + seconds + '秒';
            document.querySelector('h3').innerHTML = hTime;

        }
        setInterval(fun,1000);
    </script>

图片自动切换

    <img class="images" src="1.png" alt="">
    <div>
        <button class="btn">1</button>
        <button class="btn">2</button>
        <button class="btn">3</button>
        <button class="btn">4</button>
        <button class="btn">5</button>
    </div>


    <script>

        var arr = ['1.png','2.jpg','3.png','4.jpg','5.jpg']
        var img = document.querySelector('.images');
        var btn = document.querySelectorAll('.btn');

        var n = 1;
        for(var i = 0;i<btn.length;i++){
            btn[i].index = i;
            btn[i].onclick = function(){
                img.src = arr[this.index];
                n = this.index;
                addClass(this,'bg');
                var aArr = siblings(this);
                for(var n = 0;n<aArr.length;n++){
                    removeClass(aArr[n],'bg');
                }   

            }
            btn[i].onmouseenter = function(){
                img.src = arr[this.index];
                n = this.index;
            }   
        }
        function fun(){
            img.src = arr[n];
            n++;
            n = n === 5 ? 0 : n;//也是if表达式的意思
        // n = bool ? val1 : val2;
        }
        var t = setInterval(fun,1000);
        document.querySelector('img').onmouseenter = function(){
            clearInterval(t);
        }

        document.querySelector('img').onmouseleave = function(){
            t = setInterval(fun,1000);
        }

        function addClass(dom,str){
            var newClassName = "";
            if(dom.className){
                className = dom.className;
                newClassName = className + ' ' + str;
            }else{
                newClassName = str;
            }
            dom.className = newClassName;
        }

        function removeClass(dom,str){
            var arr = dom.className.split(" ");
            for(var i = 0;i<arr.length;i++){
                if(arr[i] === str){
                    arr.splice(i,1);
                }
            }
            dom.className = arr.join(" ");
        }

        function siblings(dom) {
            var newArr = [];
            var par = dom.parentNode;  
            var child= par.children; 
            for(var i = 0;i<child.length;i++){
                if(child[i] !== dom){
                    newArr.push(child[i]);
                }
            }
            return newArr;
        }

    </script>
